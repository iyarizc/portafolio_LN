def Traductor():
    idioma=input("A que idioma deseas traducir la frase?(Selecciona la letra) ")

    if idioma=="i":
        ingles = GoogleTranslator(source='es', target='en').translate(texto)
        print("Ingles: ",ingles)
        Traductor()
    elif idioma=="f":
        frances = GoogleTranslator(source='es', target='fr').translate(texto)
        print("Frances: ",frances)
        Traductor()
    elif idioma=="a":
        aleman = GoogleTranslator(source='es', target='de').translate(texto)
        print("Aleman: ",aleman)
        Traductor()
    elif idioma=="it":
        italiano = GoogleTranslator(source='es', target='it').translate(texto)
        print("Italiano: ",italiano)
        Traductor()
    elif idioma=="p":
        portugues = GoogleTranslator(source='es', target='pt').translate(texto)
        print("Portugues: ",portugues)
        Traductor()
    elif idioma=="x":
        exit("Gracias por usas el traductor :)")
    else:
        print("Error")
        Traductor()
Traductor()
