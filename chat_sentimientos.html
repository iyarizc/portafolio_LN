import random
from deep_translator import GoogleTranslator
from textblob import TextBlob

print("Hola, soy Pepe, tu chatbot de confianza, cuentame como te sientes... con confianza :)")

texto=input("Com贸 va todo? ").lower().strip()
ingles=GoogleTranslator(source='es', target='en').translate(texto)
blob = TextBlob(ingles)
analizar=TextBlob(ingles)
sentimiento=analizar.sentiment

print("polaridad", sentimiento.polarity)

#lista de saludosy respuestas
muy_mal=["Lamento que te sientas asi","Tranquilo, yo te apoyo",
         "Te entiendo, pero no es motivo para desanimarse"]

bien=["Que chido","Genial","Que bueno","Muy bien"]
muy_bien=["Andas de suerte","Exelente","Fantastico"]

while True:

    if sentimiento.polarity < -0.4:
        print("Chatbot:",random.choice(muy_mal))
        break
    elif sentimiento.polarity < 0 and sentimiento.polarity >= -0.4:
        print("Chatbot:",random.choice(mal))
        break
    elif sentimiento.polarity == 0:
        print("Chatbot:",random.choice(neutro))
        break
    elif sentimiento.polarity > 0 and sentimiento.polarity < 0.4:
        print("Chatbot:",random.choice(bien))
        break
    elif sentimiento.polarity > 0.4:
        print("Chatbot:",random.choice(muy_bien))
        break
